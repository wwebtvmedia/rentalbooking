FROM node:20-alpine

WORKDIR /app

COPY package.json package-lock.json ./
RUN npm ci

# copy only the app files we need for build (pages, components, styles, types, etc.)
COPY tsconfig.json ./
COPY pages ./pages
COPY components ./components
COPY styles ./styles
COPY types ./types
COPY .env .env

RUN npm run build

EXPOSE 3000

CMD ["npm", "start"]
